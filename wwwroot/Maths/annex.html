GET: Maths endpoint

List of possible query strings:

? op = + & x = number & y = number

return {"op":"+", "x":number,"y":number,"value": x + y}



? op = - & x = number & y = number

return {"op":"-", "x":number,"y":number,"value": x - y}



? op * + & x = number & y = number

return {"op":"*", "x":number,"y":number,"value": x * y}



? op = / & x = number & y = number

return {"op":"/", "x":number,"y":number,"value": x / y}



? op = % & x = number & y = number

return {"op":"%", "x":number,"y":number,"value": x % y}



? op = ! & n = number

return {"op":"%", "x":number,"value": true if n is a prime number}



? op = np & n = number

return {"op":"n", "x":number,"value": nth prime number}
